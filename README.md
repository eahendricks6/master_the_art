# master_the_art
Master the Art of Sewing Website Repo

# Steps for running locally:
# Set the npm registry:
npm config set registry https://registry.npmjs.org/

# install packages
npm install

# bulid the app 
npm run build

# serve it
npm run start

# load it
navigate to https://localhost:4200
